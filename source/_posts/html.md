### 网页基本信息

```html
<!-- 注释内容 -->

<!-- DOCTYPE 告诉浏览器我们要用什么规范 -->
<!DOCTYPE html>
<!-- 浏览器默认html 不加也没问题 -->

<html> </html>
<!-- html 总标签 -->

<head> </head>
<!-- head 网页头部 -->

<body> </body>
<!-- body 网页主体 -->
<!-- 每个网页只有一个 head , body 标签 -->

<title> </title>
<!-- title 网络标题(选项卡) -->

<!-- meta描述性标签,它用来描述网站的一些信息 -->
<!-- meta通常作SEO--搜索引擎优化 -->
<mate charset="UTF-8">
<!-- charset编码类型 -->

<mate name="keywords" contect="...">
<!-- 关键词 -->

<mate name="description" contect="...">
<!-- 描述 -->
```
