##### Matlab画图

###### 折线图

plot函数

plot(x,y,'color'//设定颜色,'r','LineStyle'//线条样式,'--')

t=0//开始:0.1//步长:10//结束;

xlabel('时间/s')//x轴说明

ylabel('振幅')//y轴说明

title('正弦函数')//标题

legend('y=sin(t)')//图表名称

hold on//开启多条曲线显示

hold off//关闭多条曲线显示

figure//创建窗口

subplot(2//行,2//列,1//第几个)//设置坐标区
